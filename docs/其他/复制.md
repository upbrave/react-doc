```tsx
import React from 'react';
import copyText from './components/copy/copyText';
import copyText2 from './components/copy/copyText2';
import copyText3 from './components/copy/copySelector';

export default () => {
  const copyHandler1 = () => {
    copyText('这是copy1的内容');
  };
  const copyHandler2 = () => {
    copyText2('这是copy2的内容');
  };
  const copyHandler3 = () => {
    copyText3('.copy_selecotr_content');
  };
  return (
    <div>
      <textarea name="" id="" cols="30" rows="2"></textarea>
      <button onClick={copyHandler1} className="copy_selecotr_content">
        复制文本1
      </button>
      <button onClick={copyHandler2}>复制文本2</button>
      <button onClick={copyHandler3}>复制标签内容</button>
    </div>
  );
};
```
